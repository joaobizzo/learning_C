5) Qual o valor impresso para a e b ?
#include <stdio.h>
int main () {
int a = 5, b = 15;
int *pA, *pB;
pA = &a;
pB = &b;
*pA = 10;
*pB = *pA;
pA = pB;
*pA = 20;
printf (“a=%d“, a);
printf(“ b=%d“, b);
getch();
}


Resposta =   A = 10 / B = 20
